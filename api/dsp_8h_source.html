463 #define dsp_buffer_stretch(buf, len, _mn, _mx)\
464 ({\
465  int k;\
466  __typeof__(buf[0]) mn = dsp_stats_min(buf, len);\
467  __typeof__(buf[0]) mx = dsp_stats_max(buf, len);\
468  double oratio = (_mx - _mn);\
469  double iratio = (mx - mn);\
470  if(iratio == 0.0) iratio = 1;\
471  for(k = 0; k < len; k++) {\
472  buf[k] -= mn;\
473  buf[k] = (__typeof__(buf[0]))((double)buf[k] * oratio / iratio);\
474  buf[k] += (__typeof__(buf[0]))_mn;\
