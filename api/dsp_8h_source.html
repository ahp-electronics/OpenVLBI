467 #define dsp_buffer_stretch(buf, len, min, max)\
468 ({\
469  int k;\
470  __typeof__(buf[0]) mn = dsp_stats_min(buf, len);\
471  __typeof__(buf[0]) mx = dsp_stats_max(buf, len);\
472  double oratio = (max - min);\
473  double iratio = (mx - mn);\
474  if(iratio == 0.0) iratio = 1;\
475  for(k = 0; k < len; k++) {\
