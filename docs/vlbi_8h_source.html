347 inline static double vlbi_default_delegate(double x, double y) {
349  return x*y;
350 }
352 inline static double vlbi_magnitude_delegate(double x, double y) {
353 return sqrt(pow(x, 2)+pow(y, 2));
354 }
356 inline static double vlbi_phase_delegate(double x, double y) {
357 double mag = sqrt(pow(x, 2)+pow(y, 2));
358 double rad = 0.0;
359 if(mag > 0.0) {
360  rad = acos (y / (mag > 0.0 ? mag : 1.0));
361  if(x < 0 && rad != 0)
362  rad = M_PI*2-rad;
363 }
364 return rad;
365 }
