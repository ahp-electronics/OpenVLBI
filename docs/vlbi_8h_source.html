652 inline double vlbi_phase_delegate(double x, double y) {
653  double mag = sqrt(pow(x, 2)+pow(y, 2));
654  double rad = 0.0;
655  if(mag > 0.0) {
656  rad = acos (y / (mag > 0.0 ? mag : 1.0));
657  if(x < 0 && rad != 0)
658  rad = PI*2-rad;
659  }
660  return rad;
661 }
